        async function validatePromo() {
            const code = document.getElementById('calc-promo').value.trim();
            if (!code) {
                currentPromoDiscount = 0;
                calculatePrice();
                return;
            }
            
            const response = await fetch('/api/promo/validate', {
                method: 'POST',
                headers: {'Content-Type': 'application/json'},
                body: JSON.stringify({code: code})
            });
            const data = await response.json();
            
            if (data.valid) {
                currentPromoDiscount = data.discount / 100;
                showToast(`✅ Промокод: скидка ${data.discount}%`);
            } else {
                currentPromoDiscount = 0;
                showToast('❌ Промокод недействителен');
            }
            calculatePrice();
        }
        
        function calculatePrice() {
            const select = document.getElementById('calc-service');
            const option = select.options[select.selectedIndex];
            if (!option.value) return;
            
            const calcType = option.getAttribute('data-type');
            const basePrice = parseFloat(option.getAttribute('data-price'));
            
            const params = {};
            document.querySelectorAll('#calc-fields input').forEach(input => {
                const name = input.id.replace('calc-', '');
                params[name] = parseFloat(input.value) || 0;
            });
            
            let price = 0;
            
            if (calcType === 'fixed') {
                price = basePrice * (params.quantity || 1);
            } else if (calcType === 'area_cm2') {
                price = (params.length / 10) * (params.width / 10) * basePrice;
            } else if (calcType === 'meter_thickness') {
                price = params.meters * (basePrice * (params.thickness / 3.0));
            } else if (calcType === 'per_minute') {
                price = params.minutes * basePrice;
            } else if (calcType === 'per_char') {
                price = params.chars * basePrice;
            } else if (calcType === 'vector_length') {
                price = params.length * basePrice;
            } else if (calcType === 'setup_batch') {
                price = (params.setup_price || basePrice) + (basePrice * (params.quantity || 1));
            } else if (calcType === 'photo_raster') {
                price = (params.length / 10) * (params.width / 10) * basePrice * (params.dpi_multiplier || 1);
            } else if (calcType === 'cylindrical') {
                price = (params.diameter * 3.14 * params.length / 100) * basePrice;
            } else if (calcType === 'volume_3d') {
                price = (params.length / 10) * (params.width / 10) * params.depth * basePrice;
            } else if (calcType === 'material_and_cut') {
                price = ((params.length / 10) * (params.width / 10) * basePrice) + (params.cut_meters * basePrice);
            }
            
            const quantity = params.quantity || 1;
            let discount = 0;
            if (quantity >= 100) discount = 20;
            else if (quantity >= 50) discount = 15;
            else if (quantity >= 20) discount = 10;
            else if (quantity >= 10) discount = 5;
            
            if (currentPromoDiscount > discount / 100) {
                discount = currentPromoDiscount * 100;
            }
            
            let finalPrice = price * (1 - discount / 100);
            let cashbackUsed = 0;
            
            const useCashback = document.getElementById('calc-cashback').value === '1';
            if (useCashback) {
                const maxCashback = finalPrice * 0.30;
                cashbackUsed = Math.min(1000, maxCashback); // Пример баланса
                finalPrice -= cashbackUsed;
            }
            
            document.getElementById('calc-total').textContent = finalPrice.toFixed(2) + '₽';
            document.getElementById('calc-discount').textContent = discount > 0 ? `Скидка ${discount}%` : '';
            document.getElementById('calc-cashback-display').textContent = cashbackUsed > 0 ? `Кэшбек: -${cashbackUsed.toFixed(2)}₽` : '';
        }

                                ${h.amount.toFixed(2)}₽ (${h.created_at})
                            </div>

        async function createPromo() {
            const code = document.getElementById('promo-code').value.trim();
            const discount = parseFloat(document.getElementById('promo-discount').value);
            const maxUses = parseInt(document.getElementById('promo-max-uses').value);
            const validUntil = document.getElementById('promo-valid-until').value;
            
            if (!code || !discount) {
                showToast('❌ Заполните код и скидку');
                return;
            }
            
            await fetch('/api/promo/create', {
                method: 'POST',
                headers: {'Content-Type': 'application/json'},
                body: JSON.stringify({
                    code: code,
                    discount: discount,
                    max_uses: maxUses || 100,
                    valid_until: validUntil || null
                })
            });
            
            showToast('✅ Промокод создан');
            loadPromoList();
            
            document.getElementById('promo-code').value = '';
            document.getElementById('promo-discount').value = '';
            document.getElementById('promo-max-uses').value = '';
            document.getElementById('promo-valid-until').value = '';
        }
